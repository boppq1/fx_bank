# 1. 자바 21 환경(조 Zulu OpenJDK)을 기본 베이스로 가져옵니다.
FROM azul/zulu-openjdk:21-slim

# 2. 컨테이너 내부 작업 디렉토리 설정
WORKDIR /app

# 3. 스프링 빌드 결과물(JAR 파일)을 컨테이너 내부로 복사합니다.
# (Gradle 빌드 시 build/libs/*.jar / Maven 빌드 시 target/*.jar)
ARG JAR_FILE=build/libs/*.jar
COPY ${JAR_FILE} app.jar

# 4. 서버를 실행할 때 적용할 환경 변수 (타임존 설정 등)
ENV TZ=Asia/Seoul

# 5. 도커 상자가 켜질 때 스프링 부트를 자동으로 실행하는 명령어
ENTRYPOINT ["java", "-jar", "app.jar"]
